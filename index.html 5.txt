    <script>
        const myPhoneNumber = "79280126881";
        const uploadedPhotos = ["PHOTO-2025-06-07-00-47-47.jpeg", "PHOTO-2025-06-07-00-54-49.jpeg", "PHOTO-2025-06-29-01-24-00 5.jpeg", "PHOTO-2025-06-29-16-03-20.jpeg", "PHOTO-2025-07-13-21-46-44.jpeg"];
        const descriptions = ["Нежный пионовый букет", "Классический подарок", "Авторская сборка", "Роскошные розы", "Цветы для настроения"];
        const products = [];
        const baseUrl = "https://raw.githubusercontent.com/6nwbbbwhpz-lgtm/----/main/";
        
        for (let i = 1; i <= 55; i++) {
            products.push({
                id: i,
                name: `Букет №${i}`,
                price: 3500 + (i % 4) * 500,
                desc: descriptions[i % descriptions.length],
                url: baseUrl + encodeURI(uploadedPhotos[i % uploadedPhotos.length])
            });
        }

        let cart = [];
        const galleryEl = document.getElementById('gallery');
        
        products.forEach(prod => {
            const card = document.createElement('div');
            card.className = 'product-card';
            card.innerHTML = `
                <img src="${prod.url}" alt="${prod.name}">
                <div class="product-title">${prod.name}</div>
                <div class="product-desc">${prod.desc}</div>
                <div class="product-price">${prod.price} руб.</div>
                <button class="add-to-cart-btn" onclick="addToCart(${prod.id})">В корзину</button>
            `;
            galleryEl.appendChild(card);
        });

        function addToCart(id) {
            const prod = products.find(p => p.id === id);
            cart.push(prod);
            updateCartUI();
        }

        function updateCartUI() {
            const items = document.getElementById('cart-items');
            items.innerHTML = cart.map(i => `<div>${i.name} — ${i.price} руб.</div>`).join('');
            document.getElementById('cart-total').innerText = 'Итого: ' + cart.reduce((sum, p) => sum + p.price, 0) + ' руб.';
        }

        function sendOrder() {
            let msg = "Здравствуйте! Хочу заказать:\n" + cart.map(i => i.name + " (" + i.price + " руб.)").join("\n");
            if(document.getElementById('extra-candy').checked) msg += "\n+ Конфеты";
            window.open(`https://wa.me/${myPhoneNumber}?text=${encodeURIComponent(msg)}`);
        }
    </script>
